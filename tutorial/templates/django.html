Django Notes

A fast guide covering project setup, apps, models, views, URLs, templates, ORM, admin, and common commands.

  • ✓ Open source and written in Python
  • ✓ Back-end + Server side + Web framework.
  • ✓ Makes it easier to build web pages.
  • ✓ Data is delivered as an Object Relational Mapping (ORM)
  • ✓ Comes with ready-to-use features like login system, database connection and CRUD operations.
  • ✓ Helpful for database driven websites
  • ✓ Follows the MVT design pattern

1) Create & Run Project

Project Setup
# install django
pip install django

# create project
django-admin startproject mysite
cd mysite

# run server
python manage.py runserver

2) Create App

# inside project root
python manage.py startapp blog

# add to settings.py
INSTALLED_APPS = [
    ...
    'blog',
]

3) Models

# blog/models.py
from django.db import models

class Post(models.Model):
    title = models.CharField(max_length=200)
    content = models.TextField()
    created = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return self.title

# apply DB changes
python manage.py makemigrations
python manage.py migrate

4) Admin

# create superuser
python manage.py createsuperuser

# blog/admin.py
from django.contrib import admin
from .models import Post
admin.site.register(Post)

5) URLs

# mysite/urls.py
from django.contrib import admin
from django.urls import path, include

urlpatterns = [
    path('admin/', admin.site.urls),
    path('', include('blog.urls')),
]

# blog/urls.py
from django.urls import path
from . import views

urlpatterns = [
    path('', views.index, name='home'),
]

6) Views

# blog/views.py
from django.shortcuts import render
from .models import Post

def index(request):
    posts = Post.objects.all()
    return render(request, 'blog/index.html', {'posts': posts})

9) Forms

# blog/forms.py
from django import forms
from .models import Post

class PostForm(forms.ModelForm):
    class Meta:
        model = Post
        fields = ['title','content']

# blog/views.py
from .forms import PostForm

def create(request):
    if request.method=='POST':
        f = PostForm(request.POST)
        if f.is_valid(): f.save()
    return render(request,'blog/form.html',{'form':f})

10) ORM Quick Reference

# SELECT
Post.objects.all()
Post.objects.filter(title__icontains='hi')
Post.objects.get(id=1)

# CREATE
Post.objects.create(title='A', content='x')

# UPDATE
p = Post.objects.get(id=1)
p.title = 'New'
p.save()

# DELETE
Post.objects.filter(id=1).delete()

12) Django REST Framework (quick)

pip install djangorestframework

# settings.py
INSTALLED_APPS += ['rest_framework']

# blog/api.py
from rest_framework import viewsets, serializers
from .models import Post

class PostSer(serializers.ModelSerializer):
    class Meta: model=Post; fields='__all__'

class PostView(viewsets.ModelViewSet):
    queryset=Post.objects.all()
    serializer_class=PostSer

# mysite/urls.py
from rest_framework import routers
from blog.api import PostView

r = routers.DefaultRouter()
r.register('posts', PostView)
urlpatterns += r.urls
